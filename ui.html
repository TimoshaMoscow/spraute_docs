UI система

Создание UI шаблона

create ui dialog_panel {
    size = [300, 128]
    background = "#F0181828"
    can_close = true
    
    text("q_text", "Нужна твоя помощь!") {
        pos = [12, 14]
        color = "#EAEAEA"
        scale = 1.0
    }
    
    button("ans_help", "Помогу") {
        pos = [12, 48]
        size = [158, 22]
        color = "#55336688"
        hover = "#66447799"
        onClick {
            chat("Выбрана помощь!")
        }
    }
    
    entity("npc:readix_avatar") {
        pos = [188, 30]
        size = [96, 98]
        scale = 1.05
        crop = [0, 0.0, 0.0, 0.4]
        anchor_y = 2.1
    }
}

Открытие и закрытие UI

# Открыть UI
ui_open(player, dialog_panel)

# Закрыть UI
ui_close(player)

# Открыть как HUD (без паузы)
overlay_open(player, hud_template)
overlay_close(player)

Обновление виджетов

# Обновление текста
ui_update(player, "my_text", "text", "Новый текст")

# Обновление цвета кнопки
ui_update(player, "my_btn", "color", "#FF0000")

# Обновление ширины
ui_update(player, "xp_fill", "w", "30%")

Анимации

# Плавное перемещение кнопки
ui_animate(player, "my_btn", "y", 150, 1.5, "bounce_out")

Динамическое создание виджетов

create ui dynamic_ui {
    size = [400, 200]
    background = "#FF000000"
    
    var i = 0
    while (i < list_size(items)) {
        val item = list_get(items, i)
        button("btn_" + i, item["name"]) {
            pos = [10, i * 24]
            size = [180, 20]
        }
        i = i + 1
    }
}

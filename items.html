Предметы и инвентарь

Основные функции

giveItem(player, item_id, [count], [name], [lore], [nbt])функция

Выдаёт предмет игроку.

giveItem(player, "minecraft:iron_ingot", 30)
giveItem(player, "minecraft:diamond", 1, "§6Королевский алмаз", ["§7Добыто в шахтах"], {"spraute_no_drop": true})
getHeldItem(player)функция

Возвращает словарь с предметом в руке. Ключи: "id", "count", "name", "slot", "lore" (список), "attack_damage", "durability" / "max_durability", "nbt".

val held = getHeldItem(player)
if (held != null && held.id == "minecraft:stick") {
    chat("У тебя палка! Урон: " + held.attack_damage)
}
player.heldItem([hand])метод

ID предмета в руке. hand: "right"/"left"/"main_hand"/"offhand".

val item = player.heldItem("right")
player.slot(n)метод

Registry id предмета в слоте. Слоты: 0-8 хотбар, 9-35 инвентарь, 36-39 броня, 40 оффхенд.

val item = player.slot(0)

Проверка предметов

# Есть ли предмет
if (player.hasItem("minecraft:iron_ingot")) {
    chat("У тебя есть железо!")
}

# Количество предметов
val count = player.countItem("minecraft:iron_ingot")

# Количество в слоте
val slot_count = player.slotCount(0)

# NBT предмета в слоте
val nbt = player.slotNbt(0)

Полная работа с инвентарём

getItemInSlot(player, slot)функция

Возвращает словарь предмета в слоте или null.

val item = getItemInSlot(player, 0)
if (item != null) {
    chat("В слоте: " + item.id)
}
getPlayerInventory(player)функция

Возвращает список словарей всех непустых слотов.

val items = getPlayerInventory(player)
for item in items {
    chat(item.id + " x" + item.count + " слот " + item.slot)
}
isSlotEmpty(player, slot)функция

Проверяет, пуст ли слот.

if (isSlotEmpty(player, 0)) {
    chat("Слот пуст")
}
hasItemInSlot(player, slot, item_id)функция

Проверяет, совпадает ли ID предмета в слоте.

if (hasItemInSlot(player, 0, "minecraft:stick")) {
    chat("В слоте палка")
}
findItemSlot(player, item_id)функция

Возвращает первый слот с предметом или -1.

val slot = findItemSlot(player, "minecraft:diamond_sword")
if (slot >= 0) {
    chat("Меч в слоте " + slot)
}
setItemInSlot(player, slot, item_id, [count], [name], [lore], [nbt])функция

Положить или заменить предмет в слоте.

setItemInSlot(player, 0, "minecraft:diamond_sword", 1, "&6Легендарный меч", ["&7Очень острый"])
setItemCount(player, slot, count)функция

Изменить количество предметов в слоте. 0 — очистить.

setItemCount(player, 0, 10)
clearItemSlot(player, slot)функция

Очистить слот.

clearItemSlot(player, 0)
removeItem(player, item_id, [count])функция

Забрать предметы у игрока. Возвращает сколько забрали.

removeItem(player, "minecraft:dirt", 64)
setItemName(player, slot, name)функция

Установить отображаемое название предмета.

setItemName(player, 0, "&6Легендарный меч")
setItemLore(player, slot, lore)функция

Установить описание предмета (список строк).

setItemLore(player, 0, ["&7Очень острый", "&c+урон"])
setItemAttackDamage(player, slot, damage)функция

Установить бонус урона предмета.

setItemAttackDamage(player, 0, 12)
setItemNbt(player, slot, nbt_dict)функция

Записать NBT-теги предмета.

setItemNbt(player, 0, {"CustomModelData": 1})
getItemNbt(player, slot)функция

Прочитать NBT-теги предмета.

val nbt = getItemNbt(player, 0)

NBT-тег spraute_no_drop

Если указать NBT-тег "spraute_no_drop": true, игрок не сможет выкинуть этот предмет из инвентаря.

giveItem(player, "minecraft:quest_item", 1, null, null, {"spraute_no_drop": true})

Кастомные орбы

# Спавн орба
val orbUuid = spawnOrb("textures/item/diamond.png", 5, x, y, z, "minecraft:overworld")

# Ожидание подбора орба игроком
await orbPickup(player, 5, "textures/item/diamond.png")

# Удаление всех орбов с текстурой
removeOrbs("textures/item/diamond.png")

Магазин на кнопку J

on keybind("j") -> shopKey {
    val player = _event_player
    
    if (player.savedData["coins"] == null) {
        player.savedData["coins"] = 50
    }
    
    val coins = player.savedData["coins"]
    
    create ui shopui {
        size = [280, 230]
        bg = "#DD101010"
        canClose = true
        
        text("title", "Магазин") {
            pos = [12, 10]
            color = "#FFD700"
            scale = 1.2
        }
        text("coinsText", "Монеты: " + intStr(coins)) {
            pos = [12, 30]
            color = "#FFFFFF"
        }
        
        scroll("shopScroll") {
            pos = [10, 55]
            size = [260, 130]
            contentH = 340
            
            item("icoApple") {
                pos = [4, 3]
                w = 16
                item = "minecraft:apple"
            }
            text("itemApple", "Яблоко x3 — 5 монет") {
                pos = [26, 5]
                color = "#FF5555"
            }
            button("buyApple", "Купить") {
                pos = [180, 2]
                size = [70, 20]
                color = "#55336688"
                hover = "#66447799"
                onClick {
                    if (player.savedData["coins"] >= 5) {
                        player.savedData["coins"] = player.savedData["coins"] - 5
                        giveItem(player, "minecraft:apple", 3)
                        playSound(player, "minecraft:entity.player.levelup", 0.6, 1.2)
                        uiUpdate(player, "coinsText", "text", "Монеты: " + intStr(player.savedData["coins"]))
                    } else {
                        playSound(player, "minecraft:entity.villager.no", 0.8, 1.0)
                    }
                }
            }
        }
        
        button("closeShop", "Закрыть") {
            pos = [168, 200]
            size = [100, 20]
            color = "#88333333"
            hover = "#AA555555"
            onClick {
                uiClose(player)
            }
        }
    }
    
    uiOpen(player, shopui)
}

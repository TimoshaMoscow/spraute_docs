Создание NPC

Базовый синтаксис

create npc readix_avatar {
    name = "ReadixGG"
    model = "geo/defolt.geo.json"
    texture = "textures/entity/defolt.png"
    pos = 8, -60, 7
    show_name = false
    speed = 0.2
    hp = 20
    collision = true
    dimension = "minecraft:overworld"
}

Свойства NPC

  • name — отображаемое имя
  • hp — здоровье и максимальное HP
  • speed — скорость движения
  • dimension — измерение для спавна
  • pos — координаты x, y, z (числа через запятую, БЕЗ скобок)
  • rotate — углы yaw, pitch (числа через запятую)
  • show_name — показывать имя (по умолчанию true)
  • collision — наличие коллизии (по умолчанию true)
  • hitbox — [ширина, высота] или [ширина, высота, ox, oy, oz]
  • hitbox_preset — "player", "small" или "large"
  • bone_hitboxes — список костных хитбоксов
  • model, texture — ресурсы модели и текстуры
  • idle_anim, walk_anim — анимации
  • drop_item — ID предмета, выпадающего при смерти
  • drop_min, drop_max — мин/макс количество дропа
  • drop_chance — шанс выпадения (0-100)

Префабы NPC

Префаб — шаблон НИПа без спавна. Один раз описываете внешность и статы, потом спавните копии с разными id и позициями.

Создание префаба

create npc_prefab goblin {
    name = "Гоблин"
    hp = 30
    model = "geo/defolt.geo.json"
    texture = "textures/entity/goblin.png"
    animation = "animations/npc_classic.animation.json"
    speed = 0.35
    dropItem = "minecraft:emerald"
    dropChance = 50
}

Спавн экземпляра по префабу

spawnNpcPrefab("goblin", "mob_01", 10, 64, 20)
spawnNpcPrefab("goblin", "mob_02", 12, 64, 22, dict("hp", 50, "name", "Берсерк"))

spawnNpcPrefab(prefabId, instanceId, x, y, z, [overrides]) — instanceId — уникальный id в NpcManager. overrides — словарь полей для подстройки.

Анимации

# Проигрывание анимации один раз
npc.playOnce("hello")

# Зацикленная анимация
npc.playLoop("sad")

# Заморозка на последнем кадре
npc.playFreeze("wave")

# Остановка анимации
npc.stop("animName")

# Сброс всех оверлейных анимаций
npc.stopOverlay()

# Процедурные анимации (дыхание, дрожание)
npc.setAdditiveWeight(1)

# Кастомная анимация смерти
npc.setDeathAnim("death")

Второй аргумент анимации: true/"additive" — накопительное смешивание, false/"replace" — замена.

Движение

# Перемещение к точке (с await блокирует до прибытия)
await npc.moveTo(5, -60, 15)

# Постоянное следование
npc.alwaysMoveTo(player)
npc.stopMove()

# Следование до расстояния
await npc.followUntil(player, 2.0)

Плавание

npc.setSwimming(true)
npc.setSwimIdleAnim("swim_idle")
npc.setSwimWalkAnim("swim_walk")

Полёт

npc.setFlying(true)
npc.flyTo(10, 80, 10, 1.2)
await npc.flyTo(20, 90, 5)
npc.alwaysFlyTo(player, 1.0)
npc.setFlying(false)

npc.setFlyIdleAnim("fly_idle")
npc.setFlyWalkAnim("fly_walk")

Взгляд

npc.lookAt(player)
npc.alwaysLookAt(player)
npc.stopLookAt()
npc.setHeadBone("head")

Хитбоксы

npc.setHitbox(0.6, 1.8)
npc.setHitbox(0.6, 1.8, 0, 0.9, 0)
npc.setHitboxOffset(0, 0.9, 0)
npc.resetHitbox()
npc.setHitboxPreset("player")

npc.addBoneHitbox("Head", 0.4, 0.4, 0.4)
npc.addBoneHitbox("hand_hit", "RightArm", 0.3, 0.5, 0.3, 0.5, 0, 0)
npc.removeBoneHitbox("hand_hit")
npc.clearBoneHitboxes()

npc.showHitboxDebug(true)

Предметы в руках

npc.setItem("right", "minecraft:iron_sword")
npc.setItem("left", "minecraft:shield", "{CustomModelData:1}")
npc.removeItem("right")
npc.removeItem()

Кастомный дроп

npc.addDrop("minecraft:iron_ingot", 1, 3, 50)
npc.drop("minecraft:apple", 5, true)

Подбор предметов

npc.pickupOnlyFrom(player)
npc.pickupAny()
val iron = npc.countItem("minecraft:iron_ingot")

Удаление NPC

npc.remove()

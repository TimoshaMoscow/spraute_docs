📦 Переменные и область видимости

Типы переменных

ТипОписаниеПример
valЛокальная переменнаяval x = 10
global valГлобальная в рамках скриптаglobal counter = 0
world valСохраняется между перезапускамиworld server_started = true
player.dataСессионные данные игрокаplayer.data["in_dialogue"] = true
player.saved_dataПостоянные данные игрокаplayer.saved_data["deaths"] = 0

Примеры использования

# Локальная переменная
val local_var = 10

# Глобальная для скрипта
global global_var = "доступна везде"

# Сохраняется между перезапусками сервера
world saved_var = 42

# Сессионные данные игрока (до перезапуска сервера)
player.data["in_dialogue"] = true

# Постоянные данные игрока (сохраняются в мире навсегда)
val p = get_player("Steve")
if (p.saved_data["deaths"] == null) {
    p.saved_data["deaths"] = 0
}
p.saved_data["deaths"] = p.saved_data["deaths"] + 1
